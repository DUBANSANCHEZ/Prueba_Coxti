DROP TABLE Persona;

CREATE TABLE Persona(
	Id_persona INT NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 1 INCREMENT BY 1),
	Nombre VARCHAR(45) NOT NULL,
	Apellido VARCHAR(45) NOT NULL,
	Cedula INT NOT NULL UNIQUE,
	Celular VARCHAR(45) NOT NULL UNIQUE,	
	Correo VARCHAR(100) NOT NULL UNIQUE,
	Recovery VARCHAR(45) NOT NULL,
	Password VARCHAR(100) NOT NULL,
	Departamento VARCHAR(45) NOT NULL,
	Ciudad VARCHAR(45) NOT NULL,
	Barrio VARCHAR(45) NOT NULL,
	Direccion VARCHAR(45) NOT NULL,
	Salario VARCHAR(45) NOT NULL,
	Otros_ingresos VARCHAR(45) NOT NULL,
	Gastos_mensuales VARCHAR(45) NOT NULL,
	Gastos_financieros VARCHAR(45) NOT NULL
);
